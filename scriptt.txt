	id_estado            serial  NOT NULL  ,
	des_estado           varchar(50)    ,
	CONSTRAINT pk_estado PRIMARY KEY ( id_estado )
 );

	id_modulo            serial  NOT NULL  ,
	CONSTRAINT pk_modulo PRIMARY KEY ( id_modulo )

	id_persona           serial  NOT NULL  ,

 );
